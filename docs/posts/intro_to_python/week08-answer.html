url = 'https://baike.baidu.com/item/%E7%99%BE%E5%BA%A6%E7%99%BE%E7%A7%91'    # 百度百科
def jumpN(n=30):
    ll = []
    tmp_url = url
    for i in range(n):
        soup = BeautifulSoup(urlopen(tmp_url).read().decode('utf-8'), features='html.parser')
        print(soup.h1.get_text())
        print(tmp_url)
        ll.append(soup.h1.get_text()+': '+tmp_url)
        all_href = [l['href'] for l in soup.find_all('a', attrs = {'target': '_blank','href' : True})]
        res = []
        for l in all_href:
            if '/item/%' in l:
                res.append('https://baike.baidu.com'+l)
        tmp_url = res[np.random.randint(len(res))]
    return ll

url_hist = jumpN(20)
